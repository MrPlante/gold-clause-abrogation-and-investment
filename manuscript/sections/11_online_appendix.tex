\begin{figure}[p]
  \caption{Industrial bond quotes. Commercial and Financial Chronicle, Oct 1931}
  \begin{center}
      \includegraphics[width=\linewidth]{figures/online-appendix/cfc_bond_quotes.png}
  \end{center}
  \label{fig:cfc_bond_quotes}
  \vspace*{3mm} \justifying \noindent
  \footnotesize{\textit{Notes.} This figure reproduces a page of industrial bond quotations from the \textit{Commercial and Financial Chronicle}, October 1931. Each row reports the coupon rate, maturity date, and recent bid--ask prices for an individual bond issue. Most bonds trade at substantial discounts to par, indicating that call options embedded in these contracts were deeply out-of-the-money, making voluntary debt restructuring prohibitively expensive for issuers during this period.}
\end{figure}

\begin{figure}[p]
  \caption{Cumulative returns around the gold clause litigation}
  \begin{center}
      \includegraphics[width=\linewidth]{figures/online-appendix/event_overview.pdf}
  \end{center}
  \label{fig:event_overview}
  \vspace*{3mm} \justifying \noindent
  \footnotesize{\textit{Notes.} This figure plots cumulative returns for the CRSP value-weighted market and the gold-exposure-weighted portfolio (with weights $w_j = d_j / \sum_k d_k$, where $d_j$ is defined in equation~(\ref{eq:d})) from March~31, 1933 through April~30, 1935, indexed to zero at the close of March~31, 1933. Vertical dashed lines mark the key dates in the episode: the suspension of the gold standard (April~19, 1933), the Joint Resolution abrogating gold clauses (June~5, 1933), the Supreme Court oral arguments (January~8, 1935), and the Supreme Court decision (February~18, 1935). Daily returns are from the same data underlying Table~\ref{tab:event_study}.}
\end{figure}
