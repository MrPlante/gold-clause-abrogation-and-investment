\begin{figure}[p]
  \caption{Industrial bond quotes. Commercial and Financial Chronicle, Oct 1931}
  \begin{center}
      \includegraphics[width=\linewidth]{Figures/Online_appendix/cfc_bond_quotes.png}
  \end{center}
  \label{fig:cfc_bond_quotes}
  \vspace*{3mm} \justifying \noindent
  \footnotesize{\textit{Notes.} This figure reproduces a page of industrial bond quotations from the \textit{Commercial and Financial Chronicle}, October 1931. Each row reports the coupon rate, maturity date, and recent bid--ask prices for an individual bond issue. Most bonds trade at substantial discounts to par, indicating that call options embedded in these contracts were deeply out-of-the-money, making voluntary debt restructuring prohibitively expensive for issuers during this period.}
\end{figure}
